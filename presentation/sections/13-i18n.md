<div class="dim" style="font-size:0.75rem;margin-bottom:0.5rem;">自定义 i18n，未用 Qt Linguist</div>

```cpp
enum class AppLanguage : std::uint8_t {
    English,
    SimplifiedChinese
};

namespace i18n {
QString tr(AppLanguage lang,
           std::string_view,
           const QString& en,
           const QString& zh) {
    return lang == SimplifiedChinese
               ? zh : en;
}
}
```

```cpp
enum class ThemeMode : std::uint8_t {
    System, Light, Dark
};

void applyApplicationTheme(
    ThemeMode mode) {
    QApplication::setStyle(
        "Fusion");
    auto pal = (resolved(mode)
                == Dark)
        ? buildDarkPalette()
        : buildLightPalette();
    QApplication::setPalette(pal);
}
```
